[−][src]Macro maplit::btreeset

macro_rules! btreeset {
    ($($key:expr,)+) => { ... };
    ( $($key:expr),* ) => { ... };
}

Create a BTreeSet from a list of elements.

Example

#[macro_use] extern crate maplit;

let set = btreeset!{"a", "b"};
assert!(set.contains("a"));
assert!(set.contains("b"));
assert!(!set.contains("c"));

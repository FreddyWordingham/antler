#[cfg(hdf5_1_12_0)]
pub const H5R_REF_BUF_SIZE: usize = 64;

#[cfg(hdf5_1_12_0)]
#[repr(C)]
#[derive(Copy, Clone)]
pub union H5R_ref_t_u {
    __data: [u8; H5R_REF_BUF_SIZE],
    align: i64,
}

#[cfg(hdf5_1_12_0)]
impl Default for H5R_ref_t_u {
    fn default() -> Self {
        unsafe { std::mem::zeroed() }
    }
}

#[cfg(hdf5_1_12_0)]
#[repr(C)]
#[derive(Copy, Clone, Default)]
pub struct H5R_ref_t {
    u: H5R_ref_t_u,
}
